39  /**
40   * Encryption class to be use for password encryption for test cases
41   * @author wf
42   *
43   */
44  @SuppressWarnings("restriction")
45  public class Crypt {
46  
47    private char[] cypher;
48    byte[] salt;
49  
50    /**
51     * @return the cypher
52     */
53    public String getCypher() {
54      return new String(cypher);
55    }
56    
57    /**
58     * return the salt
59     * @return the salt
60     */
61    public String getSalt() {
62      return new String(salt);
63    }
64  
65    /**
66     * @param cypher the cypher to set
67     */
68    public void setCypher(char[] cypher) {
69      this.cypher = cypher;
70    }
71  
72    /**
73     * create me from a password and salt
74     * 
75     * @param pCypher
76     * @param pSalt
77     */
78    Crypt(String pCypher, String pSalt) {
79      this.setCypher(pCypher.toCharArray());
80      this.salt = pSalt.getBytes();
81    }
82  
83    /**
84     * generate a Random key
85     * @param pLength
86     * @return the reandom key with the given length
87     */
88    public static String generateRandomKey(int pLength) {
89      int asciiFirst = 48;
90      int asciiLast = 122;
91      Integer[] exceptions = { 58,59,60,61,62,63,91,92,93,94,96 };
92  
93      List<Integer> exceptionsList = Arrays.asList(exceptions);
94      SecureRandom random = new SecureRandom();
95      StringBuilder builder = new StringBuilder();
96      for (int i=0; i<pLength; i++) {
97          int charIndex;
98          do {
99              charIndex = random.nextInt(asciiLast - asciiFirst + 1) + asciiFirst;
100         }
101         while (exceptionsList.contains(charIndex));
102 
103         builder.append((char) charIndex);
104     }
105 
106     return builder.toString();
107 }
108 
109   /**
110    * get a random Crypt
111    * @return a new crypt with a 32 byte random key and 8byte salt 
112    */
113   public static Crypt getRandomCrypt() {
114     String lCypher=generateRandomKey(32);
115     String lSalt=generateRandomKey(8);
116     Crypt result=new Crypt(lCypher,lSalt);
117     return result;
118   }

120   /**
121    * test this
122    * 
123    * @param args
124    * @throws Exception
125    */
126   public static void main(String[] args) throws Exception {
127     Crypt pcf=getRandomCrypt();
128     String originalPassword = "secretPassword";
129     System.out.println("Original password: " + originalPassword);
130     String encryptedPassword = pcf.encrypt(originalPassword);
131     System.out.println("Encrypted password: " + encryptedPassword);
132     String decryptedPassword = pcf.decrypt(encryptedPassword);
133     System.out.println("Decrypted password: " + decryptedPassword);
134   }

136   /**
137    * encrypt the given property
138    * @param property
139    * @return
140    * @throws GeneralSecurityException
141    * @throws UnsupportedEncodingException
142    */
143   String encrypt(String property) throws GeneralSecurityException,
144       UnsupportedEncodingException {
145     SecretKeyFactory keyFactory = SecretKeyFactory
146         .getInstance("PBEWithMD5AndDES");
147     SecretKey key = keyFactory.generateSecret(new PBEKeySpec(cypher));
148     Cipher pbeCipher = Cipher.getInstance("PBEWithMD5AndDES");
149     pbeCipher.init(Cipher.ENCRYPT_MODE, key, new PBEParameterSpec(salt, 20));
150     return base64Encode(pbeCipher.doFinal(property.getBytes("UTF-8")));
151   }
152 
153   private static String base64Encode(byte[] bytes) {
154     return new BASE64Encoder().encode(bytes);
155   }
156 
157   String decrypt(String property) throws GeneralSecurityException,
158       IOException {
159     SecretKeyFactory keyFactory = SecretKeyFactory
160         .getInstance("PBEWithMD5AndDES");
161     SecretKey key = keyFactory.generateSecret(new PBEKeySpec(cypher));
162     Cipher pbeCipher = Cipher.getInstance("PBEWithMD5AndDES");
163     pbeCipher.init(Cipher.DECRYPT_MODE, key, new PBEParameterSpec(salt, 20));
164     return new String(pbeCipher.doFinal(base64Decode(property)), "UTF-8");
165   }
166 
167   private static byte[] base64Decode(String property) throws IOException {
168     return new BASE64Decoder().decodeBuffer(property);
169   }
