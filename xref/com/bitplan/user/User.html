50    /**
51     * get the property file for the given area
52     * @param area
53     * @param area
54     * @return the property File
55     */
56    public static File getPropertyFile(String area) {
57      // get username and userhome from Operating system
58      String user = System.getProperty("user.name");
59      String userPropertiesFileName=System.getProperty("user.home") + "/."
60          + area + "/" + user + ".ini";
61      File propFile = new File(userPropertiesFileName);
62      return propFile;
63    }
64    
65    /**
66     * get the encrypted user information from the given ini area
67     * 
68     * @param area
69     *          - the name of the ini area
70     * @return the User with username and password decrypted from the ini file
71     * @throws Exception
72     */
73    public static User getUser(String area) throws Exception {
74      File propFile = getPropertyFile(area);
75      User result = new User();
76      result.props.load(new FileReader(propFile));
77      result.props = new Properties();
78      result.setUsername(result.props.getProperty("username"));
79      Crypt pcf = new Crypt(result.props.getProperty("cypher"),
80          result.props.getProperty("salt"));
81      result.setPassword(pcf.decrypt(result.props.getProperty("secret")));
82      return result;
83    }

144   /**
145    * create a credentials ini file from the command line
146    * 
147    * @param userid
148    *          - the area to create the credentials for
149    */
150   public static void createIniFile(String area) {
151     try {
152       // open up standard input
153       BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
154       if (area==null) {
155         area=getInput("area",br);
156       }
157       String username = getInput("username",br);
158       String password = getInput("password",br);
159       File propFile=getPropertyFile(area);
160       String remember= getInput("shall i store "+username+"'s credentials encrypted in "+propFile.getName()+" y/n?",br);
161       if (remember.trim().toLowerCase().startsWith("y")) {
162         Crypt lCrypt=Crypt.getRandomCrypt();
163         Properties props = new Properties();
164         props.setProperty("cypher", lCrypt.getCypher());
165         props.setProperty("salt", lCrypt.getSalt());
166         props.setProperty("user", username);
167         props.setProperty("secret", lCrypt.encrypt(password));
168         if (!propFile.getParentFile().exists()) {
169           propFile.getParentFile().mkdirs();
170         }
171         FileOutputStream propsStream=new FileOutputStream(propFile);
172         String title="Credentials for "+area;
173         props.store(propsStream, title);
174         propsStream.close();
175         System.out.println("Storing "+title+" dne.");
176       }
177     } catch (IOException e1) {
178       LOGGER.log(Level.SEVERE,e1.getMessage());
179     } catch (GeneralSecurityException e1) {
180       LOGGER.log(Level.SEVERE,e1.getMessage());
181     }
182   }
183 
184   /**

186    * 
187    * @param args
188    */
189   public static void main(String args[]) {
190     if (args.length == 0)
191       createIniFile(null);
192     else
193       createIniFile(args[0]);
194   }
