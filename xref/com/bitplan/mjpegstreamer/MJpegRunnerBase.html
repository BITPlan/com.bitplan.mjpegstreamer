168 	/**
169 	 * open the connection
170 	 * 
171 	 * @return the connection stream
172 	 */
173 	public BufferedInputStream openConnection() {
174 		BufferedInputStream result = null;
175 		try {
176 			ViewerSetting s = viewer.getViewerSetting();
177 			url = new URL(urlString);
178 			if (s.debugMode!=DebugMode.None)
179 				LOGGER.log(Level.INFO,"url: "+urlString+" readTimeOut: "+s.readTimeOut+" msecs");
180 			conn = url.openConnection();
181 			if (user != null) {
182 				String credentials = user + ":" + pass;
183 				Base64 base64 = getEncoder();
184 				byte[] encoded_credentials = base64.encode(credentials.getBytes());
185 				String authStringEnc = new String(encoded_credentials);
186 				// System.out.println("Base64 encoded auth string: " + authStringEnc);
187 				conn.setRequestProperty("Authorization", "Basic " + authStringEnc);
188 			}
189 			// change the timeout to taste, I like 1 second
190 			conn.setReadTimeout(s.readTimeOut);
191 			conn.connect();
192 			result = new BufferedInputStream(conn.getInputStream(), INPUT_BUFFER_SIZE);
193 		} catch (MalformedURLException e) {
194 			handle("Invalid URL", e);
195 		} catch (IOException ioe) {
196 			handle("Unable to connect: ", ioe);
197 		}
198 		return result;
199 	}

201 	/**
202 	 * connect
203 	 */
204 	public void connect() {
205 		connected=true;
206 		// if inputStream has been set - keep it!
207 		if (inputStream != null)
208 			return;
209 		if ("-".equals(urlString))
210 			inputStream = new BufferedInputStream(System.in, INPUT_BUFFER_SIZE);
211 		else
212 			inputStream = openConnection();
213 	}

218 	public synchronized void start() {
219 		framesReadCount=0;
220 		framesRenderedCount=0;
221 		fpscountIn=0;
222 		fpscountOut=0;
223 		if ((viewer.getViewerSetting().imageListener)!=null)
224 			this.addImageListener(viewer.getViewerSetting().imageListener);
225 		viewer.init();
226 		this.streamReader = new Thread(this, "Stream reader");
227 		streamReader.start();
228 	}

248 	/**
249 	 * handle the given exception with the given title
250 	 * 
251 	 * @param title
252 	 * @param th
253 	 */
254 	public void handle(String title, Throwable th) {
255 		String msg = title + th.getMessage();
256 		DebugMode debugMode = viewer.getViewerSetting().debugMode;
257 		switch (debugMode) {
258 		case FPS:
259 		case Verbose:
260 			LOGGER.log(Level.WARNING, msg);
261 			LOGGER.log(Level.WARNING,ExceptionUtils.getStackTrace(th));
262 			if (debugMode==DebugMode.Verbose)
263 				debugTrace(msg,this);
264 			break;
265 		default:
266 		}
267 		viewer.showMessage(msg);
268 	}

307 	/**
308 	 * read
309 	 */
310 	public boolean read() {
311 		try {
312 		  BufferedImage bufImg = MJpegHelper.getImage(curFrame);
313   		if (bufImg==null) {
314 				throw new IOException("image is null");
315 			}
316 			if (framesReadCount==0) {
317 				this.firstFrameNanoTime=System.nanoTime();
318 				this.fpsFrameNanoTime=firstFrameNanoTime;
319 				this.fpssecond=fpsFrameNanoTime;
320 			}
321 			framesReadCount++;
322 			fpscountIn++;
323 			frameAvailable = false;
324 			
325 			// uncomment next line for debug image
326 			// image= viewer.getBufferedImage("/images/start.png");
327       // viewer.repaint();
328 			// Frame per second calculation
329 			now = System.nanoTime(); 
330 			// how many nanosecs since last frame?
331 			long elapsedFrameTime = now - fpsFrameNanoTime;
332 			// how many nanosecs since last second timestamp
333 			long elapsedSecondTime = now -fpssecond;
334 			long framemillisecs = TimeUnit.MILLISECONDS.convert(elapsedFrameTime, TimeUnit.NANOSECONDS);
335 			long secmillisecs = TimeUnit.MILLISECONDS.convert(elapsedSecondTime, TimeUnit.NANOSECONDS);
336 			// is a second over?
337 			if (secmillisecs>1000) {
338 				fpsIn=fpscountIn;
339 				fpsOut=fpscountOut;
340 				fpscountOut=0;
341 				fpscountIn=0;
342 				fpssecond=now;
343 			}
344 			// do not render images that are "too quick/too early"
345 			if (framemillisecs>=this.fpsLimitMillis) {
346 				for (ImageListener listener : this.imageListeners) {
347 					if (!listener.isPostListener())
348 						listener.onRead(this, bufImg);
349 				}
350 				BufferedImage rotatedImage = this.getRotatedImage(bufImg, viewer.getViewerSetting().rotation);
351 				viewer.renderNextImage(rotatedImage);
352 				for (ImageListener listener : this.imageListeners) {
353 					if (listener.isPostListener())
354 						listener.onRead(this, rotatedImage);
355 				}
356 				// how many frames we acutally displayed
357 				framesRenderedCount++;
358 				fpsFrameNanoTime=now;
359 				fpscountOut++;
360 			}
361 
362 			switch (viewer.getViewerSetting().debugMode) {
363 			case Verbose:
364 				LOGGER.log(Level.INFO, this.getTimeMsg(" after "+framemillisecs+" msecs"));
365 				break;
366 			case FPS:
367 				if (fpssecond==now)
368 					LOGGER.log(Level.INFO, this.getTimeMsg(" after "+framemillisecs+" msecs "+fpsIn+"/"+fpsOut+" Frames per second in/out "));
369 				break;
370 			case None:
371 				break;
372 			}
373 		} catch (Throwable th) {
374 			handle("Error acquiring the frame: ", th);
375 		}
376 		return framesRenderedCount<viewer.getViewerSetting().pictureCount;
377 	}
