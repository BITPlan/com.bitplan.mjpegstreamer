36  	/**
37  	 * rotate an Image 90 Degrees to the Right
38  	 * 
39  	 * @param inputImage
40  	 * @return the rotated  image
41  	 */
42  	public static BufferedImage rotate90ToRight(BufferedImage inputImage) {
43  		int width = inputImage.getWidth();
44  		int height = inputImage.getHeight();
45  		BufferedImage returnImage = new BufferedImage(height, width,
46  				inputImage.getType());
47  
48  		for (int x = 0; x < width; x++) {
49  			for (int y = 0; y < height; y++) {
50  				returnImage.setRGB(height - y - 1, x, inputImage.getRGB(x, y));
51  			}
52  		}
53  		return returnImage;
54  	}
55  
56  	/**
57  	 * rotate 90 to Left
58  	 * 
59  	 * @param inputImage
60  	 * @return the rotated image
61  	 */
62  	public static BufferedImage rotate90ToLeft(BufferedImage inputImage) {
63  		// The most of code is same as before
64  		int width = inputImage.getWidth();
65  		int height = inputImage.getHeight();
66  		BufferedImage returnImage = new BufferedImage(height, width,
67  				inputImage.getType());
68  		// We have to change the width and height because when you rotate the image
69  		// by 90 degree, the
70  		// width is height and height is width <img
71  		// src='http://forum.codecall.net/public/style_emoticons/<#EMO_DIR#>/smile.png'
72  		// class='bbc_emoticon' alt=':)' />
73  
74  		for (int x = 0; x < width; x++) {
75  			for (int y = 0; y < height; y++) {
76  				returnImage.setRGB(y, width - x - 1, inputImage.getRGB(x, y));
77  			}
78  		}
79  		return returnImage;
80  
81  	}

83  	/**
84  	 * rotate 180
85  	 * @param inputImage
86  	 * @return the rotated image
87  	 */
88  	public static BufferedImage rotate180(BufferedImage inputImage) {
89  		if (inputImage==null)
90  		  return inputImage;
91  		// We use BufferedImage because it’s provide methods for pixel manipulation
92  		int width = inputImage.getWidth(); // the Width of the original image
93  		int height = inputImage.getHeight();// the Height of the original image
94  
95  		BufferedImage returnImage = new BufferedImage(width, height,
96  				inputImage.getType());
97  		// we created new BufferedImage, which we will return in the end of the
98  		// program
99  		// it set up it to the same width and height as in original image
100 		// inputImage.getType() return the type of image ( if it is in RBG, ARGB,
101 		// etc. )
102 
103 		for (int x = 0; x < width; x++) {
104 			for (int y = 0; y < height; y++) {
105 				returnImage.setRGB(width - x - 1, height - y - 1,
106 						inputImage.getRGB(x, y));
107 			}
108 		}
109 		// so we used two loops for getting information from the whole inputImage
110 		// then we use method setRGB by whitch we sort the pixel of the return image
111 		// the first two parametres is the X and Y location of the pixel in
112 		// returnImage and the last one is the //source pixel on the inputImage
113 		// why we put width – x – 1 and height –y – 1 is hard to explain for me, but
114 		// when you rotate image by //180degree the pixel with location [0, 0] will
115 		// be in [ width, height ]. The -1 is for not to go out of
116 		// Array size ( remember you always start from 0 so the last index is lower
117 		// by 1 in the width or height
118 		// I enclose Picture for better imagination ... hope it help you
119 		return returnImage;
120 		// and the last return the rotated image
121 
122 	}

124 	/**
125 	 * get the rotated Image
126 	 * 
127 	 * @param inputImage
128 	 * @param rotation
129 	 * @return the rotated image
130 	 */
131 	public static BufferedImage getRotatedImage(BufferedImage inputImage, int rotation) {
132 		BufferedImage result = inputImage;
133 		switch (rotation) {
134 		case 90:
135 			result = rotate90ToRight(result);
136 			break;
137 		case 180:	
138 			result = rotate180(result);
139 			break;
140 		case 270:
141 			result = rotate90ToLeft(result);
142 			break;
143 		}
144 		return result;
145 	}

147 	/**
148 	 * get the image from the given bytes
149 	 * @param imageBytes
150 	 * @return the image
151 	 * @throws IOException
152 	 */
153 	public static BufferedImage getImage(byte[] imageBytes) throws IOException {
154 		ByteArrayInputStream jpgIn = new ByteArrayInputStream(imageBytes);
155 		BufferedImage bufImg = getImage(jpgIn);
156 		return bufImg;
157 	}
158 
159 	/**
160 	 * get the Image from the given Input Stream
161 	 * @param jpgIn
162 	 * @return the image
163 	 * @throws IOException
164 	 */
165 	public static BufferedImage getImage(InputStream jpgIn) throws IOException {
166 		BufferedImage bufImg = ImageIO.read(jpgIn);
167 		jpgIn.close();
168 		return bufImg;
169 	}
