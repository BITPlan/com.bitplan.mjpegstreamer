21  package com.bitplan.mjpegstreamer;
22  
23  import java.awt.Color;
24  import java.awt.Graphics2D;
25  import java.awt.geom.Rectangle2D;
26  import java.awt.image.BufferedImage;
27  import java.util.logging.Level;
28  import java.util.logging.Logger;
29  
30  /**
31   * a listener that will draw rectangles on the image
32   * 
33   * @author wf
34   * 
35   */
36  public class RectangleOverlay implements ImageListener {
37  	private static Logger LOGGER=Logger
38  			.getLogger("com.bitplan.mjpegstreamer");
39  	
40  	static long counter;
41  	private Color color;
42  	int dwidth;
43  	int dheight;
44  	int x;
45  	int y;
46  
47  	private boolean debug=false;
48  
49  	/**
50  	 * create a RectangleOverlay based on the given parameters
51  	 * 
52  	 * @param x
53  	 * @param y
54  	 * @param dwidth
55  	 * @param dheight
56  	 * @param color
57  	 */
58  	public RectangleOverlay(int x, int y, int dwidth, int dheight, Color color) {
59  		this.color = color;
60  		this.x = x;
61  		this.y = y;
62  		this.dwidth = dwidth;
63  		this.dheight = dheight;
64  	}
65  
66  	@Override
67  	public void onRead(Object context, BufferedImage image) {
68  		Graphics2D g2d = image.createGraphics();
69  
70  		// Draw on the buffered image
71  		g2d.setColor(color);
72  		//g2d.drawRect(x, y, image.getWidth()-dwidth, image.getHeight()-dheight);
73  		int width=image.getWidth()-dwidth-x;
74  		int height=image.getHeight()-dheight-y;
75  		if (debug)
76  			LOGGER.log(Level.INFO,""+x+","+y+" "+width+"x"+height);
77  		// http://stackoverflow.com/questions/19803276/rectangle-not-drawing-on-bufferedimage
78  		g2d.draw(new Rectangle2D.Double(x,y,width,height));
79  		g2d.dispose();
80  		counter++;
81  	}
82  
83  	@Override
84  	public boolean isPostListener() {
85  		return false;
86  	}
87  
88  }
