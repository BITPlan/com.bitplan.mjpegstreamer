21  package com.bitplan.mjpegstreamer;
22  
23  import java.awt.image.BufferedImage;
24  import java.io.IOException;
25  import java.io.InputStream;
26  
27  /**
28   * runner interface
29   * @author muf
30   * http://www.gdcl.co.uk/2013/05/02/Motion-JPEG.html
31   */
32  public interface MJpegReaderRunner extends Runnable {
33  		
34  	/**
35  	 * initialize
36  	 * @param urlString
37  	 * @param user
38  	 * @param pass
39  	 * @throws IOException 
40  	 */
41  	public void init(String urlString, String user,
42  			String pass) throws IOException;
43  
44  	/**
45  	 * get the urlString that was set by init (if any)
46  	 * @return the url
47  	 */
48  	public String getUrlString();
49  	
50  	/**
51  	 * initialize
52  	 * @param viewer
53  	 * @param inputStream
54  	 * @throws IOException 
55  	 */
56  	public void init(InputStream inputStream) throws IOException;
57  	
58  	/**
59  	 * set the viewer
60  	 * @param viewer
61  	 */
62  	public void setViewer(MJpegRenderer viewer);
63  	
64  	/**
65  	 * get teh viewer
66  	 * @return the viewer
67  	 */
68  	public MJpegRenderer getViewer();
69  
70  	/**
71  	 * connect
72  	 * set the inputStream (only if it has not been set via init)
73  	 */
74  	public void connect();
75  	
76  	/**
77  	 * start
78  	 */
79  	public void start();
80  	
81  	/**
82  	 * stop me with the given message
83  	 * @param msg
84  	 */
85  	public void stop(String msg);
86  	
87  	/**
88  	 * is there a new frame?
89  	 * 
90  	 * @return whether a frame is available
91  	 */
92  	public boolean isAvailable();
93  	
94  	
95  	/**
96  	 * are we connected?
97  	 * @return whether we are connected
98  	 */
99  	public boolean isConnected();
100 
101 	/**
102 	 * get a rotated image
103 	 * @param image - the image to rotate
104 	 * @param rotation e.g. 0/90/180/270
105 	 * @return the rotated image
106 	 */
107 	public BufferedImage getRotatedImage(BufferedImage image,
108 			int rotation);
109 	
110 	/**
111 	 * read image from current frame buffer
112 	 * @return true if reading may continue
113 	 */
114 	public boolean read();
115 
116 	/**
117 	 * add an imageListener
118 	 * @param listener
119 	 */
120 	public void addImageListener(ImageListener listener);
121 	
122 	/**
123 	 * retrieve the number of milliseconds the reader has run so far
124 	 * @return the elapsed time in milliseconds
125 	 */
126 	public long elapsedTimeMillisecs();
127 	
128 	/**
129 	 * how many frames where read?
130 	 * @return the number of frames read so far
131 	 */
132 	public int getFramesRead();
133 
134 	/**
135 	 * wait until my stream is read
136 	 * @throws InterruptedException 
137 	 */
138 	public void join() throws InterruptedException;
139 
140 }
