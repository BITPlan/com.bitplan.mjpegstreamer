23  import java.io.BufferedInputStream;
24  import java.io.IOException;
25  import java.io.InputStream;
26  import java.io.StringWriter;
27  import java.net.SocketTimeoutException;
28  
29  /**
30   * Given an extended JPanel and URL read and create BufferedImages to be
31   * displayed from a MJPEG stream
32   * 
33   * @author shrub34 Copyright (c) 2012 Free for reuse, just please give me a

40  public class MJpegReaderRunner1 extends MJpegRunnerBase {
41  	private static final String CONTENT_LENGTH = "Content-length: ";
42  	private static final String CONTENT_TYPE = "Content-type: image/jpeg";
43  	private StringWriter stringWriter;
44  	
45  
46  	/**
47  	 * initialize with the given parameters
48  	 * 
49  	 * @param urlString
50  	 * @param user
51  	 * @param pass
52  	 * @throws IOException
53  	 */
54  	public void init(String urlString, String user,
55  			String pass) throws IOException {
56  		this.urlString=urlString;
57  		this.user=user;
58  		this.pass=pass;
59  		init(null);
60  	}
61  
62  	@Override
63  	public void init(InputStream inputStream)
64  			throws IOException {
65  		if (inputStream!=null)
66  			this.inputStream=new BufferedInputStream(inputStream);
67  		stringWriter = new StringWriter(128);
68  	}
69  	
70  
71  	/**
72  	 * Keeps running while process() returns true
73  	 * 
74  	 * Each loop asks for the next JPEG image and then sends it to our JPanel to
75  	 * draw
76  	 * 
77  	 * @see java.lang.Runnable#run()
78  	 */
79  	public void run() {
80  		connect();
81  		while (connected) {
82  			try {
83  				this.curFrame= retrieveNextImage();
84  				read();
85  			} catch (SocketTimeoutException ste) {
86  				handle("Lost Camera connection: " ,ste);
87  				stop("lost camera connection");
88  			} catch (IOException e) {
89  				handle("failed stream read: ", e);
90  				stop("failed stream read");
91  			}
92  		}
93  
94  		// close streams
95  		try {
96  			inputStream.close();
97  		} catch (IOException ioe) {
98  			handle("Failed to close the stream: ", ioe);
99  		}
100 	}
101 
102 	/**
103 	 * Using the <i>urlStream</i> get the next JPEG image as a byte[]
104 	 * 
105 	 * @return byte[] of the JPEG
106 	 * @throws IOException
107 	 */
108 	private byte[] retrieveNextImage() throws IOException {
109 		boolean haveHeader = false;
110 		int currByte = -1;
111 
112 		String header = null;
113 		// build headers
114 		// the DCS-930L stops it's headers
115 		while ((currByte = inputStream.read()) > -1 && !haveHeader) {
116 			stringWriter.write(currByte);
117 
118 			String tempString = stringWriter.toString();
119 			int indexOf = tempString.indexOf(CONTENT_TYPE);
120 			if (indexOf > 0) {
121 				haveHeader = true;
122 				header = tempString;
123 			}
124 		}
125 
126 		// 255 indicates the start of the jpeg image
127 		while ((inputStream.read()) != 255) {
128 			// just skip extras
129 		}
130 
131 		// rest is the buffer
132 		int contentLength = contentLength(header);
133 		byte[] imageBytes = new byte[contentLength + 1];
134 		// since we ate the original 255 , shove it back in
135 		imageBytes[0] = (byte) 255;
136 		int offset = 1;
137 		int numRead = 0;
138 		while (offset < imageBytes.length
139 				&& (numRead = inputStream.read(imageBytes, offset,
140 						imageBytes.length - offset)) >= 0) {
141 			offset += numRead;
142 		}
143 
144 		stringWriter = new StringWriter(128);
145 
146 		return imageBytes;
147 	}
148 
149 	// dirty but it works content-length parsing
150 	private static int contentLength(String header) {
151 		int indexOfContentLength = header.indexOf(CONTENT_LENGTH);
152 		int valueStartPos = indexOfContentLength + CONTENT_LENGTH.length();
153 		int indexOfEOL = header.indexOf('\n', indexOfContentLength);
154 
155 		String lengthValStr = header.substring(valueStartPos, indexOfEOL)
156 				.trim();
157 
158 		int retValue = Integer.parseInt(lengthValStr);
159 
160 		return retValue;
161 	}
162 
163 }
