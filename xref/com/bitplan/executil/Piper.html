21  package com.bitplan.executil;
22  
23  /**
24   * http://blog.art-of-coding.eu/piping-between-processes/
25   * 
26   * @author wf
27   * 
28   */
29  public class Piper implements java.lang.Runnable {
30  
31  	/**
32  	 * the two parts of the pipe,
33  	 */
34  	private java.io.InputStream input;
35  	private java.io.OutputStream output;
36  	private int bufferlen;
37  
38  	/**
39  	 * create a Piper
40  	 * 
41  	 * @param input
42  	 * @param output
43  	 * @param bufferlen
44  	 */
45  	public Piper(java.io.InputStream input, java.io.OutputStream output,
46  			int bufferlen) {
47  		this.input = input;
48  		this.output = output;
49  		this.bufferlen=bufferlen;
50  	}
51  
52  	/**
53  	 * run 
54  	 */
55  	public void run() {
56  		try {
57  			// Create bufferlen bytes buffer
58  			byte[] b = new byte[bufferlen];
59  			int read = 1;
60  			// As long as data is read; -1 means EOF
61  			while (read > -1) {
62  				// Read bytes into buffer
63  				read = input.read(b, 0, b.length);
64  				// System.out.println("read: " + new String(b));
65  				if (read > -1) {
66  					// Write bytes to output
67  					output.write(b, 0, read);
68  				}
69  			}
70  		} catch (Exception e) {
71  			// Something happened while reading or writing streams; pipe is broken
72  			throw new RuntimeException("Broken pipe", e);
73  		} finally {
74  			try {
75  				input.close();
76  			} catch (Exception e) {
77  			}
78  			try {
79  				output.close();
80  			} catch (Exception e) {
81  			}
82  		}
83  	}
84  
85  }
