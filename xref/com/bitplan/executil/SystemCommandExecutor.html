59  public class SystemCommandExecutor {
60  	public static boolean debug=true;
61  	private List<String> commandInformation;
62  	private String adminPassword;
63  	private ThreadedStreamHandler inputStreamHandler;
64  	private ThreadedStreamHandler errorStreamHandler;
65  
66  	private static final Logger logger = Logger
67  			.getLogger("com.bitplan.executil");
68  	
69  	/**
70  	 * execute the given command
71  	 * @param cmd
72  	 * @return the executor
73  	 * @throws Exception
74  	 */
75  	public static SystemCommandExecutor getExecutor(String cmd) throws Exception {
76  		List<String> commands=new ArrayList<String>();
77  		String[] parts = cmd.split(" ");
78  		for (String part:parts)
79  			commands.add(part);
80  		SystemCommandExecutor executor=new SystemCommandExecutor(commands);
81  		return executor;
82  	}
83  
84  	/**
85  	 * Pass in the system command you want to run as a List of Strings, as shown
86  	 * here:
87  	 * 
88  	 * List<String> commands = new ArrayList<String>();
89  	 * commands.add("/sbin/ping"); commands.add("-c"); commands.add("5");
90  	 * commands.add("www.google.com"); SystemCommandExecutor commandExecutor =
91  	 * new SystemCommandExecutor(commands); commandExecutor.executeCommand();
92  	 * 
93  	 * Note: I've removed the other constructor that was here to support
94  	 * executing the sudo command. I'll add that back in when I get the sudo
95  	 * command working to the point where it won't hang when the given password
96  	 * is wrong.
97  	 * 
98  	 * @param commandInformation
99  	 *            The command you want to run.
100 	 */
101 	public SystemCommandExecutor(final List<String> commandInformation) {
102 		if (commandInformation == null)
103 			throw new NullPointerException(
104 					"The commandInformation is required.");
105 		if (debug)
106 			logger.log(Level.INFO, String.valueOf(commandInformation.size()));
107 		this.commandInformation = commandInformation;
108 		this.adminPassword = null;
109 	}
110 
111 	/**
112 	 * execute the command
113 	 * @return the exit code
114 	 * @throws IOException
115 	 * @throws InterruptedException
116 	 */
117 	public int executeCommand(boolean withWait) throws IOException, InterruptedException {
118 		int exitValue = -99;
119 
120 		try {
121 			ProcessBuilder pb = new ProcessBuilder(commandInformation);
122 			Process process = pb.start();
123 
124 			// you need this if you're going to write something to the command's
125 			// input stream
126 			// (such as when invoking the 'sudo' command, and it prompts you for
127 			// a password).
128 			OutputStream stdOutput = process.getOutputStream();
129 
130 			// i'm currently doing these on a separate line here in case i need
131 			// to set them to null
132 			// to get the threads to stop.
133 			// see
134 			// http://java.sun.com/j2se/1.5.0/docs/guide/misc/threadPrimitiveDeprecation.html
135 			InputStream inputStream = process.getInputStream();
136 			InputStream errorStream = process.getErrorStream();
137 
138 			// these need to run as java threads to get the standard output and
139 			// error from the command.
140 			// the inputstream handler gets a reference to our stdOutput in case
141 			// we need to write
142 			// something to it, such as with the sudo command
143 			inputStreamHandler = new ThreadedStreamHandler(inputStream,
144 					stdOutput, adminPassword);
145 			errorStreamHandler = new ThreadedStreamHandler(errorStream);
146 
147 			// TODO the inputStreamHandler has a nasty side-effect of hanging if
148 			// the given password is wrong; fix it
149 			inputStreamHandler.start();
150 			errorStreamHandler.start();
151 
152 			if (withWait) {
153 				exitValue = process.waitFor();
154 				// Thread.sleep(50);
155 
156 				// TODO a better way to do this?
157 				inputStreamHandler.interrupt();
158 				errorStreamHandler.interrupt();
159 				inputStreamHandler.join();
160 				errorStreamHandler.join();
161 			}
162 		} catch (IOException e) {
163 			// TODO deal with this here, or just throw it?
164 			throw e;
165 		} catch (InterruptedException e) {
166 			// generated by process.waitFor() call
167 			// TODO deal with this here, or just throw it?
168 			throw e;
169 		} finally {
170 		}
171 		return exitValue;
172 	}

174 	/**
175 	 * get the Standard Input Pipe
176 	 * @return
177 	 */
178 	public  InputStream getStdInPipe() {
179 		return inputStreamHandler.getInputStream();
180 	}
181 	
182 	/**
183 	 * Get the standard output (stdout) from the command you just exec'd.
184 	 */
185 	public String getStandardOutputTextFromCommand() {
186 		return inputStreamHandler.getOutputBuffer();
187 	}
188 
189 	/**
190 	 * Get the standard error (stderr) from the command you just exec'd.
191 	 */
192 	public String getStandardErrorTextFromCommand() {
193 		return errorStreamHandler.getOutputBuffer();
194 	}
